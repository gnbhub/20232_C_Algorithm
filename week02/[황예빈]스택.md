### 풀이 과정  

c++ 에 있는 stack 을 이용하여 각 입력 명령어에 대응하는 코드를 if else 문으로 구현해주었다. 

### 소스 코드

```
#include <bits/stdc++.h>


using namespace std;

int main(){
    
    
    ios::sync_with_stdio(false); //  C와 C++ 표준 stream의 동기화를 비활성화
    cin.tie(0);

    int N; cin >> N;

    stack<int> stack;
    
    for(int i = 0; i < N; i++){

        string cmd; cin >> cmd;
        
        if(cmd == "push"){
            int val; cin >> val;
            stack.push(val);
        }
        else if(cmd == "pop"){
            if(stack.empty()){
                cout << -1 << '\n';
            }else{
                cout << stack.top() << '\n';
                stack.pop();
            }
        }else if(cmd == "top"){
            if (stack.empty()){
                cout << -1 << '\n';
            }else{
                cout << stack.top() << '\n';
            }
        }else if(cmd == "size"){
            cout << stack.size() << '\n';
            
        }else if( cmd == "empty"){
            if (stack.empty()){
                cout << 1 << '\n';
            }else{
                cout << 0 << '\n';
            }
        }

    }



}


```


#include <stdio.h>
#define MAX_SIZE 100000

int stack[MAX_SIZE];
int top = -1;

void push(int x) {
    stack[++top] = x;
}

int pop() {
    return stack[top--];
}

int main() {
    int K, x;
    int sum = 0;  

    scanf("%d", &K);
    
  

    for (int i = 0; i < K; i++) {
        scanf("%d", &x);

        if (x == 0) {
            
            sum -= pop();
        } else {
           
            push(x);
            sum += x;
        }
    }

    printf("%d\n", sum);

    return 0;
}
